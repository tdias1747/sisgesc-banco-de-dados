USE sisgesc;



-- Desativa as verificações de chave
SET FOREIGN_KEY_CHECKS = 0;

DELETE FROM fato_sisgesc;
DELETE FROM dim_aluno;
DELETE FROM dim_professor;
DELETE FROM dim_curso;
DELETE FROM dim_tempo;

-- Reativa as verificações
SET FOREIGN_KEY_CHECKS = 1;


INSERT INTO dim_aluno (nk_id_aluno, nome_aluno, status_aluno)
SELECT pk_aluno, nome_aluno, status_aluno FROM tb_aluno;

INSERT INTO dim_professor (nk_id_professor, nome_professor, titulacao)
SELECT p.pk_professor, f.nome, p.titulacao 
FROM tb_professor p 
JOIN tb_funcionario f ON p.fk_funcionario = f.pk_funcionario;

INSERT INTO dim_curso (nk_id_curso, nome_curso, modalidade)
SELECT pk_curso, nome, modalidade FROM tb_curso;

INSERT INTO dim_tempo (data_referencia, mes, ano, trimestre)
SELECT DISTINCT 
    data_pagamento, 
    MONTH(data_pagamento), 
    YEAR(data_pagamento),
    QUARTER(data_pagamento)
FROM tb_pagamento;


INSERT INTO fato_sisgesc (sk_aluno, sk_professor, sk_curso, sk_tempo, nota_media, total_faltas, valor_pago, dias_atraso)
SELECT 
    da.sk_aluno, 
    dp.sk_professor, 
    dc.sk_curso, 
    dt.sk_tempo,
    AVG(n.nota) AS nota_media, 
    IFNULL(SUM(fa.quantidade), 0) AS total_faltas, 
    SUM(p.valor_pago) AS valor_pago,
    DATEDIFF(p.data_pagamento, mens.data_vencimento) AS dias_atraso
FROM tb_matricula m
JOIN dim_aluno da ON m.fk_aluno = da.nk_id_aluno
JOIN tb_turma t ON m.fk_turma = t.pk_turma
JOIN dim_professor dp ON t.fk_professor = dp.nk_id_professor
JOIN dim_curso dc ON t.fk_curso = dc.nk_id_curso
LEFT JOIN tb_nota n ON n.fk_matricula = m.pk_matricula
LEFT JOIN tb_falta fa ON fa.fk_matricula = m.pk_matricula
JOIN tb_mensalidade mens ON mens.fk_aluno = m.fk_aluno
JOIN tb_pagamento p ON p.fk_mensalidade = mens.pk_mensalidade
JOIN dim_tempo dt ON p.data_pagamento = dt.data_referencia
GROUP BY da.sk_aluno, dp.sk_professor, dc.sk_curso, dt.sk_tempo;
