-- ÍNDICE PARA OTIMIZAR O EXPLAIN

-- ÍNDICE PARA BUSCA BUSCAS DAS NOTAS POR PROFESSOR

DROP INDEX IF EXISTS indx_fato_desempenho_prof_nota ON fato_desempenho;
CREATE INDEX  indx_fato_desempenho_prof_nota 
ON fato_desempenho (sk_professor, nota_media);

-- ÍNDICE PARA OTIMIZAR RELATÓRIOS DE INADIMPLÊNCIA
DROP INDEX IF EXISTS indx_fato_financeiro_aluno_atraso ON fato_financeiro;
CREATE INDEX indx_fato_financeiro_aluno_atraso 
ON fato_financeiro (sk_aluno, dias_atraso);

-- ÍNDICES PARA AS CHAVES ESTRANGEIRAS

DROP INDEX IF EXISTS indx_fato_desempenho_aluno ON fato_desempenho;
CREATE INDEX  indx_fato_desempenho_aluno ON fato_desempenho(sk_aluno);
DROP INDEX IF EXISTS indx_fato_frequencia_aluno ON fato_frequencia;
CREATE INDEX  indx_fato_frequencia_aluno ON fato_frequencia(sk_aluno);

-- EXPLAIN PARA A VALIDAÇÃO DOS ÍNDICES E DO BANCO DE DADOS
EXPLAIN SELECT dp.nome_professor, AVG(fd.nota_media) 
FROM fato_desempenho fd
JOIN dim_professor dp ON fd.sk_professor = dp.sk_professor
GROUP BY dp.nome_professor;
