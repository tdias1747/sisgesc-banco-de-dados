--  Carga Docente (Módulo RH está correto!)
SELECT p.nome_professor, COUNT(DISTINCT f.sk_curso) AS total_cursos, COUNT(f.sk_aluno) AS total_alunos
FROM fato_desempenho f JOIN dim_professor p ON f.sk_professor = p.sk_professor GROUP BY p.nome_professor;

-- Titulação vs Resultados (Doutores tem que render mais?)
SELECT p.titulacao, ROUND(AVG(f.nota_media), 2) AS media_notas
FROM fato_desempenho f JOIN dim_professor p ON f.sk_professor = p.sk_professor GROUP BY p.titulacao;

-- Inadimplência e Risco Financeiro
SELECT a.nome_aluno, f.valor_pago, f.dias_atraso,
CASE WHEN f.dias_atraso > 30 THEN 'CRÍTICO' WHEN f.dias_atraso > 0 THEN 'ATENÇÃO' ELSE 'OK' END AS Status
FROM fato_financeiro f JOIN dim_aluno a ON f.sk_aluno = a.sk_aluno;

-- Validação OLTP vs OLAP
SELECT 
    (SELECT SUM(valor_pago) FROM tb_pagamento) as Total_Operacional,
    (SELECT SUM(valor_pago) FROM fato_financeiro) as Total_Analitico;
