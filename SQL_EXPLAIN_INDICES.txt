-- ÍNDICES E EXPLAIN
 
-- USO DE INDEX PARA OTIMIZAR O BANCO DE DADOS


-- Limpeza e Recriação dos índices para evitar erros de duplicidade
DROP INDEX IF EXISTS indx_fato_desempenho_aluno ON fato_desempenho;
CREATE INDEX indx_fato_desempenho_aluno ON fato_desempenho(sk_aluno);

DROP INDEX IF EXISTS indx_fato_desempenho_prof ON fato_desempenho;
CREATE INDEX indx_fato_desempenho_prof  ON fato_desempenho(sk_professor);

DROP INDEX IF EXISTS indx_fato_financeiro_aluno ON fato_financeiro;
CREATE INDEX indx_fato_financeiro_aluno ON fato_financeiro(sk_aluno);

DROP INDEX IF EXISTS indx_fato_frequencia_aluno ON fato_frequencia;
CREATE INDEX indx_fato_frequencia_aluno ON fato_frequencia(sk_aluno);

-- Validação de Performance via EXPLAIN para verificar o uso dos índices
EXPLAIN SELECT dp.nome_professor, AVG(fd.nota_media)
FROM fato_desempenho fd
JOIN dim_professor dp ON fd.sk_professor = dp.sk_professor -- IRÁ EXPLANAR ALL POR CAUSA DO BAIXO NÚMERO DE DADOS NAS TABELAS DO BANCO 
GROUP BY dp.nome_professor;

EXPLAIN SELECT * FROM fato_desempenho WHERE sk_professor = 1; -- MOSTRAGEM DO RESULTADO REF PROVANDO QUE SE TIVESSE GRANDES NÚMEROS DE DADOS RESULTARIA EM ref
