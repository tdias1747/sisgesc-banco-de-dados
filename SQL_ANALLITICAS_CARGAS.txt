--  CRIAÇÃO DAS TABELAS ANALÍTICAS (VINDO DIRETAMENTE DO DIAGRAMA OLAP)

-- AS NK SÃO AS CHAVES NATURAIS QUE SE REFEREM A DADOS QUE ESTÃO PRESENTES NO MUNDO REAL E VINDO DO OLTP (DADOS QUE SÃO FAMILIARES PARA O USUÁRIO)

-- AS SK SERVEM PARA DADOS QUE NÃO ESTÃO PRESENTES NO MUNDO REAL E QUE NORMAMLMENTE VEM DO OLAP, ALÉM DE SERVIREM COMO ELO PARA QUE O FATO SE CONECTE COM A DIMENSÃO (DADOS QUE NÃO PODEM SER MUDADOS E QUE SÃO FAMILIARES COM O SISTEMA)

-- AUTO INCREMENT INSERE DADOS POSTERIORES DOS PRESENTES

-- Dimensões (Contexto Analítico)
CREATE TABLE IF NOT EXISTS dim_aluno (
    sk_aluno INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_aluno INT,
    nome_aluno VARCHAR(150),
    status_aluno VARCHAR(20)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS dim_professor (
    sk_professor INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_professor INT,
    nome_professor VARCHAR(150),
    titulacao VARCHAR(50)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS dim_curso (
    sk_curso INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_curso INT,
    nome_curso VARCHAR(200),
    modalidade VARCHAR(20)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS dim_tempo (
    sk_tempo INT AUTO_INCREMENT PRIMARY KEY,
    data_referencia DATE,
    mes INT,
    ano INT,
    trimestre INT,
    semestre VARCHAR(6)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabelas Fato (Métricas)
CREATE TABLE IF NOT EXISTS fato_desempenho (
    sk_fato INT AUTO_INCREMENT PRIMARY KEY,
    sk_aluno INT,
    sk_professor INT,
    sk_curso INT,
    sk_tempo INT,
    nota_media DECIMAL(4,2),
    FOREIGN KEY (sk_aluno)     REFERENCES dim_aluno(sk_aluno),
    FOREIGN KEY (sk_professor) REFERENCES dim_professor(sk_professor),
    FOREIGN KEY (sk_curso)     REFERENCES dim_curso(sk_curso),
    FOREIGN KEY (sk_tempo)     REFERENCES dim_tempo(sk_tempo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS fato_frequencia (
    sk_fato INT AUTO_INCREMENT PRIMARY KEY,
    sk_aluno INT,
    sk_curso INT,
    sk_tempo INT,
    total_faltas INT,
    FOREIGN KEY (sk_aluno) REFERENCES dim_aluno(sk_aluno),
    FOREIGN KEY (sk_curso) REFERENCES dim_curso(sk_curso),
    FOREIGN KEY (sk_tempo) REFERENCES dim_tempo(sk_tempo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS fato_financeiro (
    sk_fato INT AUTO_INCREMENT PRIMARY KEY,
    sk_aluno INT,
    sk_tempo INT,
    valor_pago DECIMAL(10,2),
    dias_atraso INT,
    FOREIGN KEY (sk_aluno) REFERENCES dim_aluno(sk_aluno),
    FOREIGN KEY (sk_tempo) REFERENCES dim_tempo(sk_tempo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

--  ETL - EXTRAÇÃO, TRANSFORMAÇÃO E CARGA (UTILIZAÇÃO DO FROM, INSERT E SELECT)

-- O JOIN FOI UTILZADO PARA TRAZER INFORMAÇÕES PRECISAS QUE EXISTEM REALMENTE

-- JOIN LEFT FOI UTILIZADO PARA QUE MESMO QUE NÃO POSSUA ALGO ELE FAÇA A DETERMINADA AÇÃO

-- O GROUP BY FOI UTILIZADO PARA RESUMIR DADOS MUITOS EXTENSOS E DESSES DADOS FAZER AS DEVIDAS AÇÕES

-- Limpeza total do Analítico antes da nova carga
SET FOREIGN_KEY_CHECKS = 0;
TRUNCATE TABLE fato_desempenho;
TRUNCATE TABLE fato_frequencia;
TRUNCATE TABLE fato_financeiro;
TRUNCATE TABLE dim_aluno;
TRUNCATE TABLE dim_professor;
TRUNCATE TABLE dim_curso;
TRUNCATE TABLE dim_tempo;
SET FOREIGN_KEY_CHECKS = 1;

-- Carga das Dimensões
INSERT INTO dim_aluno (nk_id_aluno, nome_aluno, status_aluno)
SELECT pk_aluno, CONCAT(primeiro_nome_aluno, ' ', sobrenome_aluno), status_aluno
FROM tb_aluno;

INSERT INTO dim_professor (nk_id_professor, nome_professor, titulacao)
SELECT p.pk_professor, CONCAT(f.primeiro_nome, ' ', f.sobrenome), p.titulacao
FROM tb_professor p
JOIN tb_funcionario f ON p.fk_funcionario = f.pk_funcionario;

INSERT INTO dim_curso (nk_id_curso, nome_curso, modalidade)
SELECT pk_curso, nome, modalidade
FROM tb_curso;

-- DIMENSÃO TEMPO EXTRAÇÃO DE DADOS BASEADA NO TEMPO DE PAGAMENTO

INSERT INTO dim_tempo (data_referencia, mes, ano, trimestre, semestre)
SELECT DISTINCT
    data_pagamento,
    MONTH(data_pagamento),
    YEAR(data_pagamento),
    QUARTER(data_pagamento),
    CONCAT(YEAR(data_pagamento), '/', CEIL(MONTH(data_pagamento) / 6))
FROM tb_pagamento;

-- Carga da Fato Desempenho
INSERT INTO fato_desempenho (sk_aluno, sk_professor, sk_curso, sk_tempo, nota_media)
SELECT da.sk_aluno, dp.sk_professor, dc.sk_curso, 1, ROUND(AVG(n.nota), 2)
FROM tb_nota n
JOIN dim_aluno    da ON n.fk_aluno       = da.nk_id_aluno
JOIN tb_turma      t ON n.fk_turma       = t.pk_turma
JOIN dim_professor dp ON t.fk_professor   = dp.nk_id_professor
JOIN dim_curso    dc ON t.fk_curso       = dc.nk_id_curso
GROUP BY da.sk_aluno, dp.sk_professor, dc.sk_curso;

-- Carga da Fato Frequência
INSERT INTO fato_frequencia (sk_aluno, sk_curso, sk_tempo, total_faltas)
SELECT da.sk_aluno, dc.sk_curso, 1, SUM(fa.quantidade)
FROM tb_falta fa
JOIN dim_aluno da ON fa.fk_aluno  = da.nk_id_aluno
JOIN tb_turma   t ON fa.fk_turma  = t.pk_turma
JOIN dim_curso dc ON t.fk_curso   = dc.nk_id_curso
GROUP BY da.sk_aluno, dc.sk_curso;

-- Carga da Fato Financeiro ( CÁLCULO PARA DIFERENCIAR DATA DE PAGAMENTO COM O VENCIMENTO)

-- O COALESCE COLOCA VALORES CASO NÃO ENCONTRE ALGO QUE PROCURA (EX:1) 

INSERT INTO fato_financeiro (sk_aluno, sk_tempo, valor_pago, dias_atraso)
SELECT
    da.sk_aluno,
    COALESCE(dt.sk_tempo, 1),
    SUM(p.valor_pago),
    SUM(CASE
        WHEN p.data_pagamento > mens.data_vencimento
        THEN DATEDIFF(p.data_pagamento, mens.data_vencimento)
        ELSE 0
    END)
FROM tb_mensalidade mens
JOIN dim_aluno  da ON mens.fk_aluno        = da.nk_id_aluno
JOIN tb_pagamento p ON p.fk_mensalidade    = mens.pk_mensalidade
LEFT JOIN dim_tempo dt ON p.data_pagamento = dt.data_referencia
GROUP BY da.sk_aluno, dt.sk_tempo;
