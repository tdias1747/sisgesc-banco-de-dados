-- CONSULTAS OLTP (TRANSACIONAL)

-- SELECT PARA LISTAR TODOS OS ALUNOS

SELECT primeiro_nome_aluno, sobrenome_aluno, cpf_aluno, status_aluno
FROM tb_aluno;

-- SUBSELECT PARA VISUALIZAÇÃO DE ALUNOS ACIMA DA MÉDIA

SELECT a.primeiro_nome_aluno, a.sobrenome_aluno
FROM tb_aluno a
WHERE a.pk_aluno IN (
    SELECT m.fk_aluno
    FROM tb_mensalidade m
    WHERE m.valor > (SELECT AVG(valor) FROM tb_mensalidade)
);

-- CONTROLE TRANSACIONAL (ACID)



-- ROLLBACK - SIMULA UM ERRO E NÃO GARANTE A INSERÇÃO DESSES DADOS NO BANCO

START TRANSACTION;

ROLLBACK;
