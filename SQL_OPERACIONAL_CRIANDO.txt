
-- DESATIVA CHECAGEM DAS CHAVES

SET FOREIGN_KEY_CHECKS = 0;

-- LIMPEZA TOTAL UTILIZANDO O DROP SE TIVER VALOR ELE SERÁ EXCLUÍDO 
DROP TABLE IF EXISTS tb_carga_horaria_docente, tb_vinculos_professor_disciplina, tb_professor, tb_funcionario;
DROP TABLE IF EXISTS tb_inadimplencia, tb_contas_receber, tb_pagamento, tb_mensalidade, tb_contrato_educacional;
DROP TABLE IF EXISTS tb_horario_grade, tb_falta, tb_nota, tb_matricula, tb_turma, tb_disciplina, tb_curso, tb_aluno;
DROP TABLE IF EXISTS fato_desempenho, fato_frequencia, fato_financeiro, dim_aluno, dim_professor, dim_curso, dim_tempo;

-- REATIVAR A CHECAGEM DAS CHAVES
SET FOREIGN_KEY_CHECKS = 1;


-- CRIAÇÃO DAS TABELAS DO OPERACIONAL (CRIANDO TABELA DO OPERACIONAL, vindo do DIAGRAMA oltp)

-- PK É UMA CHAVE ÚNICA QUE SERVE PARA MELHORAR A A PROCURA DE ALGO EM UMA TABELA (MELHORIA NO DESEMPENHO)

-- O AUTO INCREMENT GERA UM VALOR DEPOIS QUE VOCÊ INSERE UM DADO

-- O NOT NULL UNIQUE GARANTE VALOR NO CAMPO E DADOS ÚNICOS.

-- CHECK SERVE PARA CHECAR UMA OPÇÃO ANTES DELA SER APLICADA


CREATE TABLE tb_aluno (
  pk_aluno INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nome_aluno VARCHAR(150) NOT NULL,
  cpf_aluno VARCHAR(11) NOT NULL UNIQUE,
  email_aluno VARCHAR(200) NOT NULL UNIQUE,
  data_nascimento_aluno DATE NOT NULL,
  telefone_aluno VARCHAR(20),
  status_aluno VARCHAR(20) DEFAULT 'Ativo' CHECK (status_aluno IN ('Ativo', 'Trancado', 'Formado', 'Desligado')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_curso (
  pk_curso INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(200) NOT NULL,
  modalidade VARCHAR(20) NOT NULL CHECK (modalidade IN ('Presencial', 'EAD')),
  carga_horaria_curso INT NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_disciplina (
  pk_disciplina INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_curso INT NOT NULL,
  nome VARCHAR(200) NOT NULL,
  carga_horaria INT NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_curso) REFERENCES tb_curso(pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_funcionario (
  pk_funcionario INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(150) NOT NULL,
  cpf VARCHAR(11) NOT NULL UNIQUE,
  cargo VARCHAR(100) NOT NULL,
  salario DECIMAL(10,2) NOT NULL CHECK (salario > 0),
  data_admissao DATE NOT NULL,
  status_func VARCHAR(20) DEFAULT 'Ativo',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_professor (
  pk_professor INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_funcionario INT NOT NULL UNIQUE,
  titulacao VARCHAR(50) NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_funcionario) REFERENCES tb_funcionario(pk_funcionario)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_turma (
  pk_turma INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_disciplina INT NOT NULL,
  fk_professor INT NOT NULL,
  fk_curso INT NOT NULL,
  semestre VARCHAR(6) NOT NULL,
  ano INT NOT NULL,
  vaga_max INT DEFAULT 40,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_disciplina) REFERENCES tb_disciplina(pk_disciplina),
  FOREIGN KEY (fk_professor) REFERENCES tb_professor(pk_professor),
  FOREIGN KEY (fk_curso) REFERENCES tb_curso(pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_matricula (
  pk_matricula INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  fk_turma INT NOT NULL,
  data_matricula DATE DEFAULT (CURRENT_DATE),
  status_matr VARCHAR(20) DEFAULT 'Ativa',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno),
  FOREIGN KEY (fk_turma) REFERENCES tb_turma(pk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_nota (
  pk_nota INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_matricula INT NOT NULL,
  nota DECIMAL(4,2) NOT NULL CHECK (nota BETWEEN 0 AND 10),
  tipo_avaliacao VARCHAR(50),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_matricula) REFERENCES tb_matricula(pk_matricula)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_falta (
  pk_falta INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_matricula INT NOT NULL,
  quantidade INT NOT NULL,
  data_falta DATE,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_matricula) REFERENCES tb_matricula(pk_matricula)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_horario_grade (
  pk_grade INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_turma INT NOT NULL,
  dia_semana VARCHAR(15),
  horario_inicio TIME,
  horario_fim TIME,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_turma) REFERENCES tb_turma(pk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_contrato_educacional (
  pk_contrato INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  fk_curso INT NOT NULL,
  valor_total DECIMAL(10,2) NOT NULL,
  status_finan VARCHAR(20) DEFAULT 'Vigente',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno),
  FOREIGN KEY (fk_curso) REFERENCES tb_curso(pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_mensalidade (
  pk_mensalidade INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_contrato INT NOT NULL,
  fk_aluno INT NOT NULL,
  valor DECIMAL(10,2) NOT NULL,
  data_vencimento DATE NOT NULL,
  status_mensal VARCHAR(20) DEFAULT 'Pendente',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_contrato) REFERENCES tb_contrato_educacional(pk_contrato),
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_pagamento (
  pk_pagamento INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_mensalidade INT NOT NULL,
  data_pagamento DATE NOT NULL,
  valor_pago DECIMAL(10,2) NOT NULL,
  forma_pagamento VARCHAR(50) DEFAULT 'Cartão',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade(pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_contas_receber (
  pk_conta INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_mensalidade INT NOT NULL,
  valor_devido DECIMAL(10,2),
  status_conta VARCHAR(20) DEFAULT 'Aberto',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade(pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_inadimplencia (
  pk_inadimplencia INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  fk_mensalidade INT NOT NULL,
  dias_atraso INT DEFAULT 0,
  valor_multa DECIMAL(10,2) DEFAULT 0.00,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno),
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade(pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_vinculos_professor_disciplina (
  pk_vinculo INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_professor INT NOT NULL,
  fk_disciplina INT NOT NULL,
  status_vinculo VARCHAR(20) DEFAULT 'Ativo',
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_professor) REFERENCES tb_professor(pk_professor),
  FOREIGN KEY (fk_disciplina) REFERENCES tb_disciplina(pk_disciplina)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_carga_horaria_docente (
  pk_carga INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_professor INT NOT NULL,
  horas_aula INT NOT NULL,
  horas_pesquisa INT DEFAULT 0,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_professor) REFERENCES tb_professor(pk_professor)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
