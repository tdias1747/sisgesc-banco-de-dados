-- Tabelas Operacionais
CREATE TABLE tb_aluno (
  pk_aluno int NOT NULL AUTO_INCREMENT,
  nome_aluno varchar(150) NOT NULL,
  cpf_aluno varchar(11) NOT NULL UNIQUE,
  email_aluno varchar(200) NOT NULL UNIQUE,
  data_nascimento_aluno date NOT NULL,
  status_aluno varchar(20) DEFAULT 'Ativo',
  PRIMARY KEY (pk_aluno)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_funcionario (
  pk_funcionario int NOT NULL AUTO_INCREMENT,
  nome varchar(150) NOT NULL,
  cargo varchar(100) NOT NULL,
  PRIMARY KEY (pk_funcionario)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_professor (
  pk_professor int NOT NULL AUTO_INCREMENT,
  fk_funcionario int NOT NULL,
  titulacao varchar(50) DEFAULT NULL,
  PRIMARY KEY (pk_professor),
  FOREIGN KEY (fk_funcionario) REFERENCES tb_funcionario (pk_funcionario)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_curso (
  pk_curso int NOT NULL AUTO_INCREMENT,
  nome varchar(200) NOT NULL,
  modalidade varchar(20) NOT NULL,
  PRIMARY KEY (pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_turma (
  pk_turma int NOT NULL AUTO_INCREMENT,
  fk_professor int NOT NULL,
  fk_curso int NOT NULL,
  PRIMARY KEY (pk_turma),
  FOREIGN KEY (fk_professor) REFERENCES tb_professor (pk_professor),
  FOREIGN KEY (fk_curso) REFERENCES tb_curso (pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_matricula (
  pk_matricula int NOT NULL AUTO_INCREMENT,
  fk_aluno int NOT NULL,
  fk_turma int NOT NULL,
  PRIMARY KEY (pk_matricula),
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno (pk_aluno),
  FOREIGN KEY (fk_turma) REFERENCES tb_turma (pk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_nota (
  pk_nota int NOT NULL AUTO_INCREMENT,
  fk_matricula int NOT NULL,
  nota decimal(4,2) NOT NULL,
  PRIMARY KEY (pk_nota),
  FOREIGN KEY (fk_matricula) REFERENCES tb_matricula (pk_matricula)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_falta (
  pk_falta int NOT NULL AUTO_INCREMENT,
  fk_matricula int NOT NULL,
  quantidade int NOT NULL,
  PRIMARY KEY (pk_falta),
  FOREIGN KEY (fk_matricula) REFERENCES tb_matricula (pk_matricula)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_mensalidade (
  pk_mensalidade int NOT NULL AUTO_INCREMENT,
  fk_aluno int NOT NULL,
  valor decimal(10,2) NOT NULL,
  data_vencimento date NOT NULL,
  PRIMARY KEY (pk_mensalidade),
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno (pk_aluno)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_pagamento (
  pk_pagamento int NOT NULL AUTO_INCREMENT,
  fk_mensalidade int NOT NULL,
  data_pagamento date NOT NULL,
  valor_pago decimal(10,2) NOT NULL,
  PRIMARY KEY (pk_pagamento),
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade (pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Dimensões Conformadas
CREATE TABLE dim_aluno (
    sk_aluno INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_aluno INT,
    nome_aluno VARCHAR(150),
    status_aluno VARCHAR(20)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dim_professor (
    sk_professor INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_professor INT,
    nome_professor VARCHAR(150),
    titulacao VARCHAR(50)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dim_curso (
    sk_curso INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_curso INT,
    nome_curso VARCHAR(200),
    modalidade VARCHAR(20)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dim_tempo (
    sk_tempo INT AUTO_INCREMENT PRIMARY KEY,
    data_referencia DATE,
    mes INT,
    ano INT,
    trimestre INT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- FATO 1: Desempenho Acadêmico
CREATE TABLE fato_desempenho (
    sk_aluno INT,
    sk_professor INT,
    sk_curso INT,
    sk_tempo INT,
    nota_media DECIMAL(4,2),
    FOREIGN KEY (sk_aluno) REFERENCES dim_aluno(sk_aluno),
    FOREIGN KEY (sk_professor) REFERENCES dim_professor(sk_professor),
    FOREIGN KEY (sk_curso) REFERENCES dim_curso(sk_curso),
    FOREIGN KEY (sk_tempo) REFERENCES dim_tempo(sk_tempo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- FATO 2: Frequência
CREATE TABLE fato_frequencia (
    sk_aluno INT,
    sk_curso INT,
    sk_tempo INT,
    total_faltas INT,
    FOREIGN KEY (sk_aluno) REFERENCES dim_aluno(sk_aluno),
    FOREIGN KEY (sk_curso) REFERENCES dim_curso(sk_curso),
    FOREIGN KEY (sk_tempo) REFERENCES dim_tempo(sk_tempo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- FATO 3: Financeiro
CREATE TABLE fato_financeiro (
    sk_aluno INT,
    sk_tempo INT,
    valor_pago DECIMAL(10,2),
    dias_atraso INT,
    FOREIGN KEY (sk_aluno) REFERENCES dim_aluno(sk_aluno),
    FOREIGN KEY (sk_tempo) REFERENCES dim_tempo(sk_tempo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

SELECT p.pk_professor, f.nome, p.titulacao FROM tb_professor p JOIN tb_funcionario f ON p.fk_funcionario = f.pk_funcionario;

INSERT INTO dim_curso (nk_id_curso, nome_curso, modalidade) 
SELECT pk_curso, nome, modalidade FROM tb_curso;

INSERT INTO dim_tempo (data_referencia, mes, ano, trimestre) 
SELECT DISTINCT data_pagamento, MONTH(data_pagamento), YEAR(data_pagamento), QUARTER(data_pagamento) FROM tb_pagamento;

-- Carga Fato Desempenho
INSERT INTO fato_desempenho (sk_aluno, sk_professor, sk_curso, sk_tempo, nota_media)
SELECT da.sk_aluno, dp.sk_professor, dc.sk_curso, dt.sk_tempo, AVG(n.nota)
FROM tb_matricula m
JOIN dim_aluno da ON m.fk_aluno = da.nk_id_aluno
JOIN tb_turma t ON m.fk_turma = t.pk_turma
JOIN dim_professor dp ON t.fk_professor = dp.nk_id_professor
JOIN dim_curso dc ON t.fk_curso = dc.nk_id_curso
JOIN tb_nota n ON n.fk_matricula = m.pk_matricula
JOIN dim_tempo dt ON dt.sk_tempo = 1
GROUP BY da.sk_aluno, dp.sk_professor, dc.sk_curso, dt.sk_tempo;

-- Carga Fato Frequência
INSERT INTO fato_frequencia (sk_aluno, sk_curso, sk_tempo, total_faltas)
SELECT da.sk_aluno, dc.sk_curso, dt.sk_tempo, SUM(fa.quantidade)
FROM tb_matricula m
JOIN dim_aluno da ON m.fk_aluno = da.nk_id_aluno
JOIN tb_turma t ON m.fk_turma = t.pk_turma
JOIN dim_curso dc ON t.fk_curso = dc.nk_id_curso
JOIN tb_falta fa ON fa.fk_matricula = m.pk_matricula
JOIN dim_tempo dt ON dt.sk_tempo = 1
GROUP BY da.sk_aluno, dc.sk_curso, dt.sk_tempo;

-- Carga Fato Financeiro
INSERT INTO fato_financeiro (sk_aluno, sk_tempo, valor_pago, dias_atraso)
SELECT da.sk_aluno, dt.sk_tempo, SUM(p.valor_pago), DATEDIFF(p.data_pagamento, mens.data_vencimento)
FROM tb_mensalidade mens
JOIN dim_aluno da ON mens.fk_aluno = da.nk_id_aluno
JOIN tb_pagamento p ON p.fk_mensalidade = mens.pk_mensalidade
JOIN dim_tempo dt ON p.data_pagamento = dt.data_referencia
GROUP BY da.sk_aluno, dt.sk_tempo;

CREATE INDEX idx_desempenho_nota ON fato_desempenho(nota_media);
CREATE INDEX idx_financeiro_atraso ON fato_financeiro(dias_atraso);

-- Validação de Explain (Características do Código)

EXPLAIN SELECT p.nome_professor, AVG(f.nota_media) 
FROM fato_desempenho f 
JOIN dim_professor p ON f.sk_professor = p.sk_professor 
GROUP BY p.nome_professor;


-- BI 1: Correlação Titulação vs Notas (Pág 39 do Word p/ teste)
SELECT p.titulacao, ROUND(AVG(f.nota_media), 2) as media_notas
FROM fato_desempenho f JOIN dim_professor p ON f.sk_professor = p.sk_professor GROUP BY p.titulacao;

-- BI 2: Risco de Inadimplência (Pág 38 do WORD p/ teste)
SELECT a.nome_aluno, f.dias_atraso,
CASE WHEN f.dias_atraso > 30 THEN 'CRÍTICO' WHEN f.dias_atraso > 0 THEN 'ATENÇÃO' ELSE 'EM DIA' END as Status
FROM fato_financeiro f JOIN dim_aluno a ON f.sk_aluno = a.sk_aluno;
