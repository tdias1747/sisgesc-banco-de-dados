-- CRIAÇÃO DO OPERACIONAL (OLTP) - Módulo Académico

-- UTILIZAÇÃO DO CHECK PARA PROTEÇÃO DE DADOS (NÃO OCORRE ERRO, CASO OCORRA ELE RETORNA ERRO)

-- UTILIZAÇÃO DO CURRENT_TIMESTAMP PARA JUSTAMENTE TER O TEMPO REAL

-- A UTILIZAÇÃO DA FOREIGN KEY E REFERENCES PARA NÃO TER DADOS ÓRFÃOS E CRIAÇÃO DE VÍNCULO (PAI E FILHA)

-- NOT NULL OBRIGANDO COLOCAR VALORES NOS CAMPOS

-- UNIQUE RESEPRENTANDO VALORES ÚNICOS NOS CAMPOS

-- PK UTILIZAÇÃO DE CHAVES PRIMÁRIAS (ORDENAÇÃO DO BANCO DE DADOS)

-- FK UTILIZAÇÃO DE CHAVES SECUNDÁRIAS PARA A CRIAÇÃO VÍNCULOS COM OUTRAS TABELAS

CREATE TABLE tb_aluno (
  pk_aluno INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  primeiro_nome_aluno VARCHAR(50) NOT NULL,
  sobrenome_aluno VARCHAR(100) NOT NULL,
  cpf_aluno VARCHAR(11) NOT NULL UNIQUE,
  data_nascimento_aluno DATE NOT NULL,
  status_aluno VARCHAR(20) DEFAULT 'Ativo' CHECK (status_aluno IN ('Ativo', 'Trancado', 'Formado', 'Desligado')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_email_aluno (
  pk_email INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  endereco_email VARCHAR(200) NOT NULL UNIQUE,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_telefone_aluno (
  pk_telefone INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  numero_telefone VARCHAR(20) NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_curso (
  pk_curso INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nome_curso VARCHAR(200) NOT NULL,
  modalidade_curso VARCHAR(20) NOT NULL CHECK (modalidade_curso IN ('Presencial', 'EAD')),
  carga_horaria_curso DECIMAL(6,2) NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_disciplina (
  pk_disciplina INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_curso INT NOT NULL,
  nome_disciplina VARCHAR(200) NOT NULL,
  carga_horaria_disciplina DECIMAL(6,2) NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_curso) REFERENCES tb_curso(pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Módulo RH (CRIAÇÃO DAS TABELAS RELACIONADAS AO RH)

CREATE TABLE tb_funcionario (
  pk_funcionario INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  primeiro_nome_funcionario VARCHAR(50) NOT NULL,
  sobrenome_funcionario VARCHAR(100) NOT NULL,
  cpf_funcionario VARCHAR(11) NOT NULL UNIQUE,
  cargo_funcionario VARCHAR(100) NOT NULL,
  salario_funcionario DECIMAL(10,2) NOT NULL CHECK (salario_funcionario > 0),
  data_admissao_funcionario DATE NOT NULL,
  status_funcionario VARCHAR(20) DEFAULT 'Ativo' CHECK (status_funcionario IN ('Ativo', 'Inativo', 'Demitido', 'Afastado')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_professor (
  pk_professor INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_funcionario INT NOT NULL UNIQUE,
  titulacao_professor VARCHAR(50) NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_funcionario) REFERENCES tb_funcionario(pk_funcionario)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_vinculos_professor_disciplina (
  fk_professor INT NOT NULL,
  fk_disciplina INT NOT NULL,
  status_vinculo VARCHAR(20) DEFAULT 'Ativo' CHECK (status_vinculo IN ('Ativo', 'Inativo')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (fk_professor, fk_disciplina),
  FOREIGN KEY (fk_professor) REFERENCES tb_professor(pk_professor),
  FOREIGN KEY (fk_disciplina) REFERENCES tb_disciplina(pk_disciplina)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_carga_horaria_docente (
  pk_carga INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_professor INT NOT NULL,
  horas_aula DECIMAL(6,2) NOT NULL,
  horas_pesquisa DECIMAL(6,2) DEFAULT 0.00,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_professor) REFERENCES tb_professor(pk_professor)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Módulo Academico (2 PARTE) (REPRESENTANDO A CRIAÇÃO DAS TABELAS DO MÓDULO ACADÊMICO)

CREATE TABLE tb_turma (
  pk_turma INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_disciplina INT NOT NULL,
  fk_professor INT NOT NULL,
  fk_curso INT NOT NULL,
  semestre_turma VARCHAR(6) NOT NULL,
  ano_turma INT NOT NULL,
  vagas_max_turma INT DEFAULT 40,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_disciplina) REFERENCES tb_disciplina(pk_disciplina),
  FOREIGN KEY (fk_professor) REFERENCES tb_professor(pk_professor),
  FOREIGN KEY (fk_curso) REFERENCES tb_curso(pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_matricula (
  fk_aluno INT NOT NULL,
  fk_turma INT NOT NULL,
  data_matricula DATE DEFAULT (CURRENT_DATE),
  status_matricula VARCHAR(20) DEFAULT 'Ativa' CHECK (status_matricula IN ('Ativa', 'Trancada', 'Cancelada', 'Concluida')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (fk_aluno, fk_turma),
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno),
  FOREIGN KEY (fk_turma) REFERENCES tb_turma(pk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_nota (
  fk_aluno INT NOT NULL,
  fk_turma INT NOT NULL,
  tipo_avaliacao VARCHAR(50) NOT NULL,
  valor_nota DECIMAL(4,2) NOT NULL CHECK (valor_nota BETWEEN 0 AND 10),
  data_avaliacao DATE DEFAULT (CURRENT_DATE),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (fk_aluno, fk_turma, tipo_avaliacao),
  FOREIGN KEY (fk_aluno, fk_turma) REFERENCES tb_matricula(fk_aluno, fk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_falta (
  fk_aluno INT NOT NULL,
  fk_turma INT NOT NULL,
  data_falta DATE NOT NULL,
  quantidade_falta INT NOT NULL DEFAULT 1,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (fk_aluno, fk_turma, data_falta),
  FOREIGN KEY (fk_aluno, fk_turma) REFERENCES tb_matricula(fk_aluno, fk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_horario_grade (
  pk_grade INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_turma INT NOT NULL,
  dia_semana_grade VARCHAR(15) CHECK (dia_semana_grade IN ('Segunda', 'Terca', 'Quarta', 'Quinta', 'Sexta', 'Sabado', 'Domingo')),
  horario_inicio_grade TIME NOT NULL,
  horario_fim_grade TIME NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_turma) REFERENCES tb_turma(pk_turma)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Módulo Financeiro (REPRESENTANDO A CRIAÇÃO DAS TABELAS FINANCEIRAS)

CREATE TABLE tb_contrato_educacional (
  pk_contrato INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  fk_curso INT NOT NULL,
  valor_total_contrato DECIMAL(10,2) NOT NULL,
  status_contrato VARCHAR(20) DEFAULT 'Vigente' CHECK (status_contrato IN ('Vigente', 'Encerrado', 'Cancelado')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno),
  FOREIGN KEY (fk_curso) REFERENCES tb_curso(pk_curso)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_mensalidade (
  pk_mensalidade INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_contrato INT NOT NULL,
  fk_aluno INT NOT NULL,
  valor_mensalidade DECIMAL(10,2) NOT NULL,
  data_vencimento_mensalidade DATE NOT NULL,
  status_mensalidade VARCHAR(20) DEFAULT 'Pendente' CHECK (status_mensalidade IN ('Pendente', 'Pago', 'Atrasado', 'Negociado')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_contrato) REFERENCES tb_contrato_educacional(pk_contrato),
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_pagamento (
  pk_pagamento INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_mensalidade INT NOT NULL,
  data_pagamento DATE NOT NULL,
  valor_pago DECIMAL(10,2) NOT NULL,
  forma_pagamento VARCHAR(50) DEFAULT 'Cartão' CHECK (forma_pagamento IN ('Cartao', 'Boleto', 'PIX', 'Dinheiro')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade(pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_contas_receber (
  pk_conta INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_mensalidade INT NOT NULL,
  valor_devido DECIMAL(10,2) NOT NULL,
  status_conta VARCHAR(20) DEFAULT 'Aberto' CHECK (status_conta IN ('Aberto', 'Pago', 'Cancelado')),
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade(pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tb_inadimplencia (
  pk_inadimplencia INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_aluno INT NOT NULL,
  fk_mensalidade INT NOT NULL,
  data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  ultima_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (fk_aluno) REFERENCES tb_aluno(pk_aluno),
  FOREIGN KEY (fk_mensalidade) REFERENCES tb_mensalidade(pk_mensalidade)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- CONSULTAS DAS TABELAS OPERACIONAIS OLTP TRANSACIONAL (LISTAGEM DE ALUNOS E COMPARAÇÃO ENTRE ALUNOS)

SELECT primeiro_nome_aluno, sobrenome_aluno, cpf_aluno, status_aluno FROM tb_aluno;

SELECT a.primeiro_nome_aluno, a.sobrenome_aluno
FROM tb_aluno a
WHERE a.pk_aluno IN (
    SELECT m.fk_aluno
    FROM tb_mensalidade m
    WHERE m.valor_mensalidade > (SELECT AVG(valor_mensalidade) FROM tb_mensalidade)
);

-- CRIAÇÃO DAS TABELAS ANALÍTICAS (VINDO DO DIAGRAMA OLAP)

-- VARCHAR SENDO O MÁXIMO DE CARACTERES POSSÍVEL

-- AS NK SENDO AS CHAVES NATURAIS QUE LIGAM OS DADOS DO OPERACIONAL E REPRESENTA OS DADOS REAIS (COM CARACTERÍSTICAS DO CRIADOR)

-- AS SK REPRESENTANDO DADOS QUE NÃO SÃO DO MUNDO REAL E SIM DO PRÓPRIO AMBIENTE DO BANCO DE DADOS E REFERENCIA DADOS DAS TABELAS ANALITICAS, ALÉM DE CAUSAR O VÍNCULO DA FATO COM A DIM

CREATE TABLE IF NOT EXISTS dim_aluno (
    sk_aluno INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_aluno INT,
    nome_aluno VARCHAR(150),
    status_aluno VARCHAR(20)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS dim_professor (
    sk_professor INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_professor INT,
    nome_professor VARCHAR(150),
    titulacao VARCHAR(50)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS dim_curso (
    sk_curso INT AUTO_INCREMENT PRIMARY KEY,
    nk_id_curso INT,
    nome_curso VARCHAR(200),
    modalidade VARCHAR(20)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS dim_tempo (
    sk_tempo INT AUTO_INCREMENT PRIMARY KEY,
    data_referencia DATE,
    mes INT,
    ano INT,
    trimestre INT,

-- O JOIN FOI UTILIZADO PARA REPRESENTAR INFORMAÇÕES PRECISAS QUE REALMENTE EXISTEM

-- JOIN LEFT FOI UTILIZADO PARA QUE MESMO QUE NÃO POSSUA ALGO ELE FAÇA A DETERMINADA AÇÃO

-- O GROUP BY FOI UTILIZADO PARA RESUMIR DADOS EXTENSOS E DESSES DADOS FAZER AS DEVIDAS AÇÕES

-- CARGA DAS TABELAS DE DIMENSÕES E DA FATO ( CÁLCULO PARA DIFERENCIAR DATA DE PAGAMENTO COM O VENCIMENTO,  OU SEJA RELACIONANDO O VENCIMENTO DOS PAGAMENTOS)

-- O COALESCE COLOCA VALORES CASO NÃO ENCONTRE ALGO QUE PROCURA (EX:1)

-- UTILIZAÇÃO DO FROM PARA REALIZAÇÃO DE DADOS CORRETAMENTE


-- CARGA NA TABELA DIMENSIONADA ALUNO E CONCATENANDO O NOME DO ALUNO

INSERT INTO dim_aluno (nk_id_aluno, nome_aluno, status_aluno)
SELECT pk_aluno, CONCAT(primeiro_nome_aluno, ' ', sobrenome_aluno), status_aluno FROM tb_aluno;


-- CARGA NA TABELA DIMENSIONADA PROFESSOR E CONCATENANDO OS VALORES RETIRADOS DA TABELA PROFESSOR

INSERT INTO dim_professor (nk_id_professor, nome_professor, titulacao)
SELECT p.pk_professor, CONCAT(f.primeiro_nome_funcionario, ' ', f.sobrenome_funcionario), p.titulacao_professor
FROM tb_professor p JOIN tb_funcionario f ON p.fk_funcionario = f.pk_funcionario;

-- INSERINDO CARGA NA TABELA DIMENSIONADA CURSO
INSERT INTO dim_curso (nk_id_curso, nome_curso, modalidade)
SELECT pk_curso, nome_curso, modalidade_curso FROM tb_curso;


-- FAZENDO A INSERÇÃO DE CARGAS PEGANDO TODAS AS CARACTERÍSTICAS RELACIONADAS AO TEMPO EM OUTRAS TABELAS ( SERVINDO DE BASE PARA O CÁLCULO DE VENCIMENTO DO PAGAMENTO)

INSERT INTO dim_tempo (data_referencia, mes, ano, trimestre, semestre)
SELECT DISTINCT data_pagamento, MONTH(data_pagamento), YEAR(data_pagamento), QUARTER(data_pagamento), 
CONCAT(YEAR(data_pagamento), '/', CEIL(MONTH(data_pagamento) / 6)) FROM tb_pagamento;


-- CÁLCULO DO DESEMPENHO DO ALUNO (ANALISANDO AS TABELAS CURSO, PROFESSOR...)

INSERT INTO fato_desempenho (sk_aluno, sk_professor, sk_curso, sk_tempo, nota_media)
SELECT da.sk_aluno, dp.sk_professor, dc.sk_curso, 1, ROUND(AVG(n.valor_nota), 2)
FROM tb_nota n
JOIN dim_aluno da ON n.fk_aluno = da.nk_id_aluno
JOIN tb_turma t ON n.fk_turma = t.pk_turma
JOIN dim_professor dp ON t.fk_professor = dp.nk_id_professor
JOIN dim_curso dc ON t.fk_curso = dc.nk_id_curso
GROUP BY da.sk_aluno, dp.sk_professor, dc.sk_curso;

-- CÁLCULO SOBRE A FREQUÊNCIA DO ALUNO (ANALISANDO TABELAS DE TEMPO, CURSO...)

INSERT INTO fato_frequencia (sk_aluno, sk_curso, sk_tempo, total_faltas)
SELECT da.sk_aluno, dc.sk_curso, 1, SUM(fa.quantidade_falta)
FROM tb_falta fa
JOIN dim_aluno da ON fa.fk_aluno = da.nk_id_aluno
JOIN tb_turma t ON fa.fk_turma = t.pk_turma
JOIN dim_curso dc ON t.fk_curso = dc.nk_id_curso
GROUP BY da.sk_aluno, dc.sk_curso;


-- CARGA NA FATO FINANCEIRO SOBRE O CÁLCULO DE VENCIMENTO OU DIAS DE VENCIMENTOS DE PAGAMENTO QUE O ALUNO POSSUI

INSERT INTO fato_financeiro (sk_aluno, sk_tempo, valor_pago, dias_atraso)
SELECT da.sk_aluno, COALESCE(dt.sk_tempo, 1), SUM(p.valor_pago),
SUM(CASE WHEN p.data_pagamento > mens.data_vencimento_mensalidade THEN DATEDIFF(p.data_pagamento, mens.data_vencimento_mensalidade) ELSE 0 END)
FROM tb_mensalidade mens
JOIN dim_aluno da ON mens.fk_aluno = da.nk_id_aluno
JOIN tb_pagamento p ON p.fk_mensalidade = mens.pk_mensalidade
LEFT JOIN dim_tempo dt ON p.data_pagamento = dt.data_referencia
GROUP BY da.sk_aluno, dt.sk_tempo;


-- USO DE INDEX PARA A OTIMIZAÇÃO DOS DADOS (PARA MELHORAR A CAPACIDADE DO BANCO DE DADOS APLICANDO OS INDX NAS TABELAS)

DROP INDEX IF EXISTS indx_fato_desempenho_aluno ON fato_desempenho;
CREATE INDEX indx_fato_desempenho_aluno ON fato_desempenho(sk_aluno);

DROP INDEX IF EXISTS indx_fato_desempenho_prof ON fato_desempenho;
CREATE INDEX indx_fato_desempenho_prof ON fato_desempenho(sk_professor);

DROP INDEX IF EXISTS indx_fato_financeiro_aluno ON fato_financeiro;
CREATE INDEX indx_fato_financeiro_aluno ON fato_financeiro(sk_aluno);

DROP INDEX IF EXISTS indx_fato_frequencia_aluno ON fato_frequencia;
CREATE INDEX indx_fato_frequencia_aluno ON fato_frequencia(sk_aluno);


-- Validação DA Performance com EXPLAIN 

EXPLAIN SELECT dp.nome_professor, AVG(fd.nota_media)
FROM fato_desempenho fd
JOIN dim_professor dp ON fd.sk_professor = dp.sk_professor -- IRÁ EXPLANAR ALL POR CAUSA DO BAIXO NÚMERO DE DADOS NAS TABELAS DO BANCO 
GROUP BY dp.nome_professor;

EXPLAIN SELECT * FROM fato_desempenho WHERE sk_professor = 1; -- MOSTRAGEM DO RESULTADO REF PROVANDO QUE SE TIVESSE GRANDES NÚMEROS DE DADOS RESULTARIA EM ref


-- UTILIZAÇÃO DOS BIs + VALIDAÇÃO OLTP E OLAP (RESULTADO SERÁ OS DOIS TENDO O MESMO RESULTADO)

-- BI 1: Comparativo entre as Notas e a Titulação do devido Professor 

SELECT p.titulacao, ROUND(AVG(f.nota_media), 2) AS media_notas, COUNT(*) AS qtd_avaliacoes
FROM fato_desempenho f JOIN dim_professor p ON f.sk_professor = p.sk_professor 
GROUP BY p.titulacao ORDER BY media_notas DESC;

-- BI 2: Risco de Inadimplência (Análise de inadimplência)

SELECT a.nome_aluno, f.valor_pago, f.dias_atraso AS dias_atraso_total,
CASE WHEN f.dias_atraso > 30 THEN 'CRÍTICO' WHEN f.dias_atraso > 0 THEN 'ATENÇÃO' ELSE 'EM DIA' END AS status_inadimplencia
FROM fato_financeiro f JOIN dim_aluno a ON f.sk_aluno = a.sk_aluno ORDER BY f.dias_atraso DESC;

-- BI 3: Validação OLTP e OLAP (AMOSTRAGEM DA IGUALDADE ENTRE ELES)
SELECT 
    (SELECT ROUND(SUM(valor_pago), 2) FROM tb_pagamento) AS Total_Operacional_OLTP,
    (SELECT ROUND(SUM(valor_pago), 2) FROM fato_financeiro) AS Total_Analitico_OLAP,
    ROUND((SELECT SUM(valor_pago) FROM fato_financeiro) - (SELECT SUM(valor_pago) FROM tb_pagamento), 2) AS Diferenca;
